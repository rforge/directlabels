Make text bounding box larger by some amount.

enlarge.box <- function(d,...){
  if(!"h"%in%names(d))stop("need to have already calculated height and width.")
  h <- unit(d$h,"native")
  d$h <- d$h*2
  d$w <- d$w+as.numeric(convertWidth(convertHeight(h,"inches"),"native"))
  calc.borders(d)
}
