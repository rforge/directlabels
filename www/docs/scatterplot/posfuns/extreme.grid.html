directlabels - scatterplot - Positioning Function - extreme.grid

Use empty.grid with extreme.points.

extreme.grid <- empty.grid.fun(extreme.points)

cylinders

data(mpg,package="ggplot2")
m <- lm(cty~displ,data=mpg)
mpgf <- fortify(m,mpg)
p <- mpg.scatter <- xyplot(.resid~.fitted,mpgf,groups=factor(cyl))
direct.label(p,"extreme.grid")
